    import torch.nn as nn
    from torchvision.models import resnet18, ResNet18_Weights

    def build_frozen_resnet18() -> nn.Module:
        """
        Returns a ResNet18 with:
          - ImageNet pretrained weights (ResNet18_Weights.IMAGENET1K_V1)
          - Final fc layer replaced with nn.Identity()
          - All parameters frozen (requires_grad=False)
          - eval mode (so BatchNorm uses running stats, not batch stats)
        Output shape: (batch, 512)
        """
        model = resnet18(weights=ResNet18_Weights.IMAGENET1K_V1)
        model.fc = nn.Identity()
        for p in model.parameters():
            p.requires_grad = False
        model.eval()
        return model

IMPORTANT: Backbone must stay in eval() mode even during training (for BatchNorm).
The adapter/head can be trained; backbone is just a feature extractor.

--- TASK 5.1: Create src/adapter.py ---
WHAT: Bottleneck adapter module.
WHERE: src/adapter.py
HOW:

    import torch
    import torch.nn as nn

    class BottleneckAdapter(nn.Module):
        """
        Bottleneck adapter: Down-project to rank r, ReLU, Up-project back to dim.
        Residual connection is added by the caller (or here - see below).
        Parameter count: dim*r + r + r*dim + dim = 2*dim*r + dim + r
        For dim=512, r=16: ~16,912 trainable parameters.
        """
        def __init__(self, dim: int, rank: int):
            super().__init__()
            self.down = nn.Linear(dim, rank)
            self.act = nn.ReLU(inplace=True)
            self.up = nn.Linear(rank, dim)
            # Initialize up-projection to near-zero so adapter starts as identity
            nn.init.zeros_(self.up.weight)
            nn.init.zeros_(self.up.bias)

        def forward(self, x: torch.Tensor) -> torch.Tensor:
            return x + self.up(self.act(self.down(x)))

IMPORTANT: The up-projection is initialized to zero. This makes the adapter
           start as an identity function (output = input), so at step 0 the
           model behaves exactly like frozen-ResNet18 + head. This is a
           standard trick from the Adapter paper (Houlsby et al. 2019).

--- TASK 6.1: Create src/model.py ---
WHAT: Combine backbone + adapter + head into one nn.Module.
WHERE: src/model.py
HOW:

    import torch
    import torch.nn as nn
    from .backbone import build_frozen_resnet18
    from .adapter import BottleneckAdapter

    class BPEFTModel(nn.Module):
        """
        Full model:
          x -> [frozen ResNet18] -> features (B, 512)
            -> [Bottleneck Adapter] -> adapted features (B, 512)
            -> [Linear head] -> logits (B, num_classes)

        For EVIDENTIAL mode: apply ReLU to logits to get non-negative evidence.
        For SOFTMAX mode: return raw logits (softmax is inside the loss fn).
        """
        def __init__(self, num_classes: int, feature_dim: int = 512,
                     adapter_rank: int = 16, mode: str = "evidential"):
            super().__init__()
            assert mode in ("evidential", "softmax")
            self.mode = mode
            self.backbone = build_frozen_resnet18()
            self.adapter = BottleneckAdapter(feature_dim, adapter_rank)
            self.head = nn.Linear(feature_dim, num_classes)

        def forward(self, x: torch.Tensor) -> torch.Tensor:
            with torch.no_grad():
                feats = self.backbone(x)   # (B, 512), frozen
            feats = self.adapter(feats)    # (B, 512), trainable
            logits = self.head(feats)      # (B, num_classes)
            if self.mode == "evidential":
                # Evidence must be non-negative
                return torch.relu(logits)
            return logits

TASK 7: LOSSES

--- TASK 7.1: Create src/losses.py ---
WHAT: Evidential MSE loss + KL divergence regularization. Baseline CE.
WHERE: src/losses.py
HOW: Implement three functions. Reference: the Sensoy 2018 paper and
     dougbrion/pytorch-classification-uncertainty repo. Cite this in a
     docstring comment at the top of the file.

FUNCTION A: kl_divergence_dirichlet(alpha, num_classes)
  """
  KL divergence between Dir(alpha_tilde) and uniform Dir(1,...,1).
  alpha_tilde is a "smoothed" alpha where the true-class parameter is
  clamped to 1 (so it is not penalized). In practice this is computed
  on the full alpha by the caller (see evidential_mse_loss).

  Formula (Sensoy 2018, Eq 13):
    KL[Dir(alpha) || Dir(1)] = log(Gamma(sum_k alpha_k)) - sum_k log(Gamma(alpha_k))
                             + log(Gamma(K))
                             + sum_k (alpha_k - 1) * (digamma(alpha_k) - digamma(sum_k alpha_k))
  Use torch.lgamma and torch.digamma.
  Return shape: (batch,)
  """

FUNCTION B: evidential_mse_loss(evidence, target_onehot, num_classes, kl_weight)
  """
  Evidential MSE loss from Sensoy 2018, Eq 5.
    alpha = evidence + 1
    S = sum(alpha, dim=-1, keepdim=True)
    p = alpha / S
    err_term = (target - p) ** 2
    var_term = p * (1 - p) / (S + 1)
    loss_per_sample = sum(err_term + var_term, dim=-1)

  Plus KL regularization on "wrong-class" evidence:
    alpha_tilde = target_onehot + (1 - target_onehot) * alpha   # zero out true-class
    kl = kl_divergence_dirichlet(alpha_tilde, num_classes)
    total = loss_per_sample + kl_weight * kl
  Return mean over batch.
  """

FUNCTION C: softmax_ce_loss(logits, target)
  """
  Standard cross entropy loss, for the baseline.
  """

IMPORTANT DETAIL: For the evidential model, the model outputs evidence (>= 0)
  directly. Do NOT add +1 inside the model; add +1 inside the loss function
  (alpha = evidence + 1). This is the convention in the Sensoy paper.

TASK 8: METRICS

--- TASK 8.1: Create src/metrics.py ---
WHAT: Accuracy, ECE, Brier, OOD AUROC.
WHERE: src/metrics.py
HOW: Implement four functions.

FUNCTION A: accuracy(probs, targets)
  """
  probs: (N, K) tensor, targets: (N,) long tensor.
  Return scalar top-1 accuracy in [0, 1].
  """

FUNCTION B: expected_calibration_error(probs, targets, num_bins=10)
  """
  Standard ECE:
    1. For each sample, get predicted class = argmax(probs) and confidence = max(probs).
    2. Bin samples by confidence into num_bins equal-width bins in [0, 1].
    3. For each non-empty bin, compute |avg_confidence - avg_accuracy|.
    4. Weight by bin size and sum.
  Return scalar.
  """

FUNCTION C: brier_score(probs, targets, num_classes)
  """
  Mean squared error between probs and one-hot targets.
    BS = mean over samples of sum_k (probs_k - onehot_k) ** 2
  Return scalar.
  """

FUNCTION D: ood_auroc(id_scores, ood_scores)
  """
  id_scores: higher means MORE confident (e.g., max softmax prob, or 1-vacuity).
  ood_scores: same.
  Labels: id=1, ood=0.
  Use sklearn.metrics.roc_auc_score.
  Return scalar in [0, 1].
  """

IMPORTANT: For evidential model, the "confidence" to feed into ECE/AUROC is
           p_k = alpha_k / S (the expected class probability), not evidence.
           And the OOD score is (1 - vacuity) = 1 - K/S.
           For softmax model, confidence is just softmax(logits).

================================================================================
DEBUGGING TIPS
================================================================================

Common problems and quick fixes:

1. CIFAR-100 download fails
   -> Check internet connectivity. The file is ~170MB.
   -> As a fallback, manually download from:
      https://www.cs.toronto.edu/~kriz/cifar-100-python.tar.gz

2. SVHN download fails
   -> Fallback: use Fashion-MNIST or any other dataset as OOD.
      (Update CFG.ood_dataset accordingly.)

3. Evidential loss goes NaN
   -> Check evidence is always >= 0 (ReLU applied in model).
   -> Check alpha = evidence + 1 (never 0, so lgamma is safe).
   -> Clamp evidence: torch.clamp(evidence, min=0, max=1e6).

4. Training accuracy stuck at 20% (random guess)
   -> Double-check adapter is not zeroing everything (initialization).
   -> Double-check head has requires_grad=True.
   -> Double-check optimizer is passed trainable params only (or all params
      with backbone frozen via requires_grad=False — either works).

5. ECE looks wrong
   -> Make sure probs sum to 1 per row (they should after softmax or
      alpha/S normalization).

6. OOD AUROC near 0.5 (random)
   -> Check that id_scores and ood_scores are on the correct scale.
   -> Remember: higher score should mean "more in-distribution".

7. "CUDA out of memory"
   -> Reduce batch size in config.
   -> 25 samples of 224x224 + ResNet18 fits easily on T4 (4GB is plenty).
   -> If still an issue, use batch inference in chunks for query set.
